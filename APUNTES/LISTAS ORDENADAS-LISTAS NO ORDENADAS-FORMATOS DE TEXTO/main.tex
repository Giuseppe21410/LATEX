    \item lista 1
    \item lista 2
    \item lista 3
\end{itemize}
%Para crear listas enumaras y  ordenadas, utilizamos el comando "\begin{enumerate}":
\begin{enumerate}
    \item lista 1
    \item lista 2
    \item lista 3
\end{enumerate}

% Estos son algunos formatos de texto:

% Se pueden anidar diferentes  formatos: 
\textbf{\emph{Caca de vaca}} % En negrita y cursiva {...}.
